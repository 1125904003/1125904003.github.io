# Bat命令学习

# Bat文件下的文件剪贴案列

@echo off

rem 获取当前日期
for /f "tokens=1-3 delims=/" %%d in ("%date%") do (
  set "year=%%f"
  set "month=%%e"
  set "day=%%d"
)

rem 去除月份和日期中的前导零
if "%month:~0,1%"=="0" set "month=%month:~1%"
if "%day:~0,1%"=="0" set "day=%day:~1%"

rem 创建目标目录
set "dest_folder=F:\test\file2\%month%.%day%"
mkdir "%dest_folder%"

rem 将源文件夹中的所有文件移动到新创建的目标文件夹中
set "source_folder=C:\Users\Luhuan\Desktop\file1"
move /y "%source_folder%\*" "%dest_folder%"
pause
